    std::vector<std::string> Sample::GetPngFiles(const std::string& path) {
        std::vector<std::string> pngFiles;
        if (!std::filesystem::exists(path) || !std::filesystem::is_directory(path)) {
            return pngFiles;
        }
        for (const auto& entry : std::filesystem::directory_iterator(path)) {
            if (!entry.is_regular_file()) continue;
            std::string filename = entry.path().filename().string();
            if (filename.size() < 4) continue;
            std::string ext = filename.substr(filename.size() - 4);
            std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
            if (ext == ".png") {
                pngFiles.push_back(entry.path().string()); // 返回完整路径
            }
        }
        return pngFiles;
    }
	std::vector<std::shared_ptr<Sample>> Sample::Load(const std::string& path) {
        std::vector<std::shared_ptr<Sample>> samples;
        // 检查目录是否存在
        if (!std::filesystem::exists(path) || !std::filesystem::is_directory(path)) {
            Log::Debug("Directory not found: " + path);
            return samples;
        }
        // 遍历目录中的所有文件
        for (const auto& entry : std::filesystem::directory_iterator(path)) {
            if (!entry.is_regular_file()) continue;
            std::string filepath = entry.path().string();
            std::string filename = entry.path().filename().string();
            // 只处理 .png 文件（大小写不敏感）
            if (filename.size() < 4) continue;
            std::string ext = filename.substr(filename.size() - 4);
            std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
            if (ext != ".png") continue;
            // 去掉 .png 后缀，得到 imgValue（如 "0123"）
            std::string imgValue = filename.substr(0, filename.size() - 4);
            // 加载样本
            samples.push_back(Load(filepath, imgValue));
        }
        Log::Debug("Loaded " + std::to_string(samples.size()) + " samples from " + path);
        return samples;
	}
    std::vector<std::shared_ptr<Sample>> Sample::Load(const std::vector<std::string>& filepaths) {
        std::vector<std::shared_ptr<Sample>> samples;
        for (const auto& filepath : filepaths) {
            if (!std::filesystem::exists(filepath) || !std::filesystem::is_regular_file(filepath)) {
                Log::Debug("File not found or not regular file: " + filepath);
                continue;  // 跳过无效文件，继续处理下一个
            }
            // 获取文件名（不含路径），用于提取 imgValue
            std::string filename = std::filesystem::path(filepath).filename().string();
            // 检查扩展名是否为 .png（大小写不敏感）
            if (filename.size() < 4) {
                Log::Debug("Filename too short, skipping: " + filename);
                continue;
            }
            std::string ext = filename.substr(filename.size() - 4);
            std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
            if (ext != ".png") {
                Log::Debug("Not a PNG file, skipping: " + filename);
                continue;
            }
            // 去掉 .png 后缀，得到 imgValue（如 "0123"）
            std::string imgValue = filename.substr(0, filename.size() - 4);
            // 加载样本
            auto sample = Load(filepath, imgValue);  // 假设有 Load(filepath, imgValue) 方法
            if (sample) {
                samples.push_back(sample);
            }
        }

        return samples;

	/// <summary>
	/// 处理客户端发送的包，异步执行
	/// </summary>
	/// <param name="frameType"></param>
	/// <param name="client"></param>
	/// <param name="buffer"></param>
	/// <returns></returns>
	bool WebProcess::HandelMessage(int32_t frameType, const std::shared_ptr<WebSocket>& client, const std::vector<char>& buffer) {
		if (!client) return false;
		if (frameType == Poco::Net::WebSocket::FRAME_BINARY) {
			return HandelMessageBinary(client, buffer);
		}
		return false;
	}
	bool WebProcess::SendMessageBinary(const std::shared_ptr<WebSocket>& client, const WebSocketPackage& package) {
		if (!client) return false;
		std::vector<char> buffer;
		size_t estimatedSize = sizeof(package.version) + sizeof(package.code) +
			sizeof(package.size);
		for (int32_t i = 0; i < package.size; ++i) {
			estimatedSize += sizeof(package.length) + package.length[i];  // 长度字段 + 实际数据
		}
		//预分配内存
		buffer.reserve(estimatedSize);
		//版本号
		buffer.insert(buffer.end(),
			reinterpret_cast<const char*>(&package.version),
			reinterpret_cast<const char*>(&package.version) + sizeof(package.version)); //参数二 起始指针 参数三 结束指针
		//消息类型
		buffer.insert(buffer.end(),
			reinterpret_cast<const char*>(&package.code),
			reinterpret_cast<const char*>(&package.code) + sizeof(package.code));
		//数据流总数量
		buffer.insert(buffer.end(),
			reinterpret_cast<const char*>(&package.size),
			reinterpret_cast<const char*>(&package.size) + sizeof(package.size));
		//数据流数据
		if (package.size > 0) {
			for (int32_t i = 0; i < package.size; ++i) { 
				int32_t length = package.length[i]; //包长度
				char* context = package.context[i];//包内容
				// 安全检查
				if (!context || length <= 0) {
					Log::Debug("Invalid data block.");
					return false;
				}
				//先插入长度
				buffer.insert(buffer.end(),
					reinterpret_cast<const char*>(&length),
					reinterpret_cast<const char*>(&length) + sizeof(length));
				//再插入内容
				buffer.insert(buffer.end(),
					reinterpret_cast<const char*>(context),
					reinterpret_cast<const char*>(context) + length);
			}
		}
		client->sendFrame(buffer.data(), buffer.size(), Poco::Net::WebSocket::FRAME_BINARY);
		return true;
	}

	bool WebProcess::SendMessageBinary(const std::shared_ptr<WebSocket>& client, int32_t code) {
		WebSocketPackage package;
		package.version = SERVER_VERSION;
		package.code = code;
		package.size = 0;
		package.length = nullptr;
		package.context = nullptr;
		return SendMessageBinary(client, package);
	}

	bool WebProcess::ParseHeaderBinary(const std::shared_ptr<WebSocket>& client, const std::vector<char>& buffer, WebSocketPackage& package) {
		if (buffer.size() < sizeof(package.version)) {
			MessageDebug(client,"Header Binary Data format error.The data size is incorrect.");
			return false;
		}
		int offset = 0;
		//version
		std::memcpy(&package.version, buffer.data(), sizeof(package.version));
		if (package.version != SERVER_VERSION) {
			MessageDebug(client,std::format("Server and client versions do not match,server version:{},client version:{}", SERVER_VERSION, package.version));
			return false;
		}
		offset += sizeof(package.version);
		if (buffer.size() < offset + sizeof(package.code)) {
			MessageDebug(client, "Header Binary Data format error.The data size is incorrect.");
			return false;
		}
		std::memcpy(&package.code, buffer.data() + offset, sizeof(package.code));
		offset += sizeof(package.code);
		return true;
	}

	bool WebProcess::HandelMessageBinary(const std::shared_ptr<WebSocket>& client, const std::vector<char>& buffer) {
		WebSocketPackage package;
		bool success = ParseHeaderBinary(client, buffer,package);
		if (!success) return false;
		Log::Debug(std::format("Received client message,code:{}", package.code));
		//解析Code
		switch (package.code)
		{
		case CLIENT_TO_SERVER_HEART_BEAT: { //心跳机制
			SendMessageBinary(client, SERVER_TO_CLIENT_HEART_BEAT);
		}
			break;
		default:
			break;
		}
		return true;
	}
	bool WebProcess::MessageDebug(const std::shared_ptr<WebSocket>& client, const std::string message) {
		Log::Debug(message);
		return true;
	}
}

    float Loss::Forward(const Tensor3D& input, const std::array<int32_t, 4>& target) {

# macOS 独立部署说明

本文档描述当前仓库在 Apple Silicon/macOS 上的可行独立部署路径。

## 范围

这条路径覆盖：

- `apps/frontend`
- `services/yijianvisionend`
- MySQL / Redis / MinIO / Milvus 这类本地基础依赖

这条路径明确不覆盖：

- `engines/hw-control`
- 本机 Triton / NVIDIA GPU 推理部署

也就是说，当前 macOS 方案是“前后端加基础依赖可本机运行”，不是“整套实验室硬件与 GPU 推理链路都迁到 Mac”。

## 前提

- macOS Apple Silicon
- Docker Desktop
- Node.js 18+
- Maven 3.8+
- JDK 8（推荐 Amazon Corretto 8 arm64）
- 如需导出 PDF，安装 LibreOffice

## 1. 启动基础依赖

```bash
./scripts/dev-up.sh deps macos-standalone
```

这会拉起：

- MySQL
- Redis
- etcd
- MinIO
- Milvus standalone
- Attu

不会拉起：

- Triton
- 前端容器
- 后端容器
- 硬件控制相关工程

## 2. 启动后端

```bash
./scripts/macos-standalone-backend.sh
```

脚本会自动：

- 选择 JDK 8
- 将后端指向 `127.0.0.1`
- 将日志目录设置到 `.runtime/macos-standalone/logs`
- 将上传目录设置到 `.runtime/macos-standalone/uploadPath`
- 启用 Milvus，并连接本机 compose 暴露的 `127.0.0.1:19530`

注意：

- 当前脚本不部署 Triton
- 依赖硬件或 GPU 推理的功能不在本机独立部署保障范围内
- 如果 `9090` 已被其他进程占用，可改用 `BACKEND_PORT=9190 ./scripts/macos-standalone-backend.sh`

## 3. 启动前端

```bash
cd apps/frontend
npm ci
npm run dev
```

如果后端改跑 `9190`，前端开发态也要同步覆盖 API 地址：

```bash
cd apps/frontend
VITE_API_URL=http://127.0.0.1:9190/ npm run dev
```

默认访问地址：

```text
http://127.0.0.1:5173/#/login
```

前端开发环境默认直连：

```text
http://127.0.0.1:9090/
```

本机 Milvus 健康检查地址：

```text
http://127.0.0.1:9091/healthz
```

业务流 WebSocket 默认运行时配置已改为：

```text
ws://127.0.0.1:9090/ws
```

但在当前 macOS standalone 预设里，前端默认不会主动连接这条硬件 WebSocket：

- `WEBSOCKET_ENABLED=false`
- `engines/hw-control` 不在部署范围内

如果需要生成不依赖 `/api/` 反代的构建产物：

```bash
cd apps/frontend
npm run build:standalone
```

如果后端改跑 `9190`，构建时可以直接复用同一个端口变量：

```bash
cd apps/frontend
BACKEND_PORT=9190 npm run build:standalone
```

## 4. 验证

```bash
./scripts/dev-smoke.sh macos-standalone
```

如果后端改跑 `9190`，验证命令也要同步覆盖：

```bash
BACKEND_URL=http://127.0.0.1:9190/ ./scripts/dev-smoke.sh macos-standalone
```

这个预设会：

- 检查前端根路径
- 检查后端直连接口
- 检查 MinIO 健康状态
- 检查 Milvus 健康状态
- 跳过前端 `/api/` 反代检查
- 跳过 Triton 健康检查

## 5. 已知限制

- `engines/hw-control` 不在当前部署范围内
- Triton 仍然是 GPU / NVIDIA 导向，不适合作为当前 Mac 本机依赖
- 需要 PDF 导出时，macOS 上必须能探测到 LibreOffice 的 `soffice`

- From: `yjtweb`
- To: `apps/frontend`

## Scope
- 迁移前端路由、布局、业务页面、主题与静态资源。

1. 路由切换为 Hash 模式（`/#/...`）。

## Verification Checklist
- `npm run build` 通过。
- 访问 `/#/login` 正常。
- 登录后默认进入 `/#/top-home`。
- 业务菜单可从后端成功加载。

# Visual Diff (Vue vs Axure)

## Inputs
- Pair list: `tools/visual-diff/pairs.json`
- Captures:
  - `artifacts/output/current-visual-compare/capture/vue/*.png`
  - `artifacts/output/current-visual-compare/capture/axure/*.png`

`pairs.json` now uses `?chromeless=1` for Vue URLs to hide app shell (header/top menu/side nav) and compare business content against Axure pages more fairly.

## Run diff on captured images
```bash
python3 tools/visual-diff/run_compare_from_capture.py \
  --pairs tools/visual-diff/pairs.json \
  --capture-root artifacts/output/current-visual-compare/capture \
  --report-dir artifacts/output/current-visual-compare/report \
  --tolerance 20 \
  --threshold 0.03
```

Outputs:
- Per page: `artifacts/output/current-visual-compare/report/<key>.metrics.json`
- Summary: `artifacts/output/current-visual-compare/report/summary.json`

## Notes
- The diff tool uses macOS `sips` to decode PNG into BMP and compares pixels with `numpy`.
- If screenshots have different sizes, comparison uses the overlap area (top-left aligned).

# DeepLR Backend

Spring Boot backend for the DeepLR neural network learning project.

## Requirements

- Java 8 or newer

This project includes Maven Wrapper, so global Maven is not required.

## Run

```bash
.\mvnw.cmd spring-boot:run
```

## Test

```bash
.\mvnw.cmd test
```

## Dataset Storage

The backend creates dataset folders under `DEEPLR_DATA_DIR`.

If the environment variable is not set, it defaults to:

```text
${user.dir}
```

Current synthetic digit dataset folder:

```text
datasets/
  synthetic-digits/
```

# Expert Review Project Storage

## Scope
- 2D expert review project snapshots are stored in `expert_review_project.project_json`.
- The snapshot payload now persists overlap fusion color state under `compare2d.overlap.fusionColorize`.

## Snapshot Shape
```json
{
  "compare2d": {
    "overlap": {
      "topSide": "right",
      "operationSide": "left",
      "effectiveInteractionSide": "left",
      "leftOpacity": 1,
      "rightOpacity": 0.5,
      "fusionColorize": {
        "enabled": true,
        "fusionColor": "#808080"
```
